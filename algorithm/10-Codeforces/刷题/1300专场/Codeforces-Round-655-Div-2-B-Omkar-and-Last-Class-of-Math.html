using namespace std;

bool is_prime(int x) {
    for (int i = 2; i * i <= x; i++) {
        if (x % i == 0) return false;
    }
    return true;
}

void f() {
    for (int i = 3; i <= 10000; i += 2) {
        if (is_prime(i))continue;
        int res = INT_MAX;
        int x, y;
        for (int j = 1; j < i; j++) {
            int t = lcm(j, i - j);
            if (t < res) {
                res = t;
                x = j, y = i - j;
            }
        }
        cout << i << " " << x << " " << y << endl;
    }
}

void solve() {
    int n;
    cin >> n;
    if (n % 2 == 0) {
        cout << n / 2 << ' ' << n / 2 << endl;
    } else {
        if (is_prime(n)) {
            cout << 1 << " " << n - 1 << endl;
        } else {
            for (int k = 2;; k++) {
                if (n % (k + 1) == 0) {
                    int x=n/(k+1);
                    int y=x*k;
                    cout<<x<<' '<<y<<endl;
                    return;
                }
            }
        }
    }
}
